## List Members

<api>GET /api/v1/members</api>

List all registered members (built-in and custom).

<ParamField query="type" type="string">
  Filter by member type: `Think`, `Function`, `Data`, `API`
</ParamField>

<ParamField query="builtin" type="boolean">
  Filter to only built-in members
</ParamField>

<ParamField query="page" type="number" default="1">
  Page number for pagination
</ParamField>

<ParamField query="limit" type="number" default="50">
  Items per page (max 100)
</ParamField>

### Response

<ResponseField name="members" type="array">
  Array of member objects

  <ResponseField name="members[].name" type="string">
    Member name
  </ResponseField>

  <ResponseField name="members[].type" type="string">
    Member type: `Think`, `Function`, `Data`, `API`
  </ResponseField>

  <ResponseField name="members[].builtin" type="boolean">
    Whether this is a built-in member
  </ResponseField>

  <ResponseField name="members[].description" type="string">
    Member description
  </ResponseField>

  <ResponseField name="members[].version" type="string">
    Member version
  </ResponseField>
</ResponseField>

<ResponseField name="pagination" type="object">
  Pagination metadata

  <ResponseField name="pagination.page" type="number">
    Current page
  </ResponseField>

  <ResponseField name="pagination.limit" type="number">
    Items per page
  </ResponseField>

  <ResponseField name="pagination.total" type="number">
    Total number of members
  </ResponseField>

  <ResponseField name="pagination.pages" type="number">
    Total number of pages
  </ResponseField>
</ResponseField>

<RequestExample>

```bash cURL (All Members)
curl https://your-worker.workers.dev/api/v1/members \
  -H "Authorization: Bearer YOUR_API_KEY"
```

```bash cURL (Filter by Type)
curl "https://your-worker.workers.dev/api/v1/members?type=Think" \
  -H "Authorization: Bearer YOUR_API_KEY"
```

```bash cURL (Built-in Only)
curl "https://your-worker.workers.dev/api/v1/members?builtin=true" \
  -H "Authorization: Bearer YOUR_API_KEY"
```

```javascript JavaScript
const response = await fetch(
  'https://your-worker.workers.dev/api/v1/members',
  {
    headers: {
      'Authorization': 'Bearer YOUR_API_KEY'
    }
  }
);

const { members } = await response.json();
console.log(`Found ${members.length} members`);
```

```typescript TypeScript (SDK)
import { ConductorClient } from '@ensemble-edge/conductor/sdk';

const client = new ConductorClient({
  baseURL: 'https://your-worker.workers.dev',
  apiKey: process.env.CONDUCTOR_API_KEY
});

const members = await client.listMembers();
console.log(members);
```

</RequestExample>

## Get Member Details

<api>GET /api/v1/members/:name</api>

Get detailed information about a specific member.

<ParamField path="name" type="string" required>
  Member name
</ParamField>

### Response

<ResponseField name="name" type="string">
  Member name
</ResponseField>

<ResponseField name="type" type="string">
  Member type
</ResponseField>

<ResponseField name="builtin" type="boolean">
  Whether built-in
</ResponseField>

<ResponseField name="description" type="string">
  Member description
</ResponseField>

<ResponseField name="version" type="string">
  Member version
</ResponseField>

<ResponseField name="config" type="object">
  Configuration schema

  <ResponseField name="config.schema" type="object">
    JSON schema for configuration
  </ResponseField>

  <ResponseField name="config.defaults" type="object">
    Default configuration values
  </ResponseField>
</ResponseField>

<ResponseField name="input" type="object">
  Input schema

  <ResponseField name="input.schema" type="object">
    JSON schema for input
  </ResponseField>

  <ResponseField name="input.required" type="array">
    Required input fields
  </ResponseField>
</ResponseField>

<ResponseField name="output" type="object">
  Output schema

  <ResponseField name="output.schema" type="object">
    JSON schema for output
  </ResponseField>
</ResponseField>

<ResponseField name="examples" type="array">
  Usage examples
</ResponseField>

<RequestExample>

```bash cURL
curl https://your-worker.workers.dev/api/v1/members/scrape \
  -H "Authorization: Bearer YOUR_API_KEY"
```

```javascript JavaScript

```

## Member Statistics

<api>GET /api/v1/members/:name/stats</api>

Get usage statistics for a member.

<ParamField path="name" type="string" required>
  Member name
</ParamField>

<ParamField query="period" type="string" default="24h">
  Time period: `1h`, `24h`, `7d`, `30d`
</ParamField>

### Response

<ResponseField name="name" type="string">
  Member name
</ResponseField>

<ResponseField name="period" type="string">
  Time period
</ResponseField>

<ResponseField name="executions" type="number">
  Total executions
</ResponseField>

<ResponseField name="successes" type="number">
  Successful executions
</ResponseField>

<ResponseField name="failures" type="number">
  Failed executions
</ResponseField>

<ResponseField name="avgDuration" type="number">
  Average duration in milliseconds
</ResponseField>

<ResponseField name="p95Duration" type="number">
  95th percentile duration
</ResponseField>

<ResponseField name="cacheHitRate" type="number">
  Cache hit rate (0-1)
</ResponseField>

<RequestExample>

```bash cURL
curl "https://your-worker.workers.dev/api/v1/members/scrape/stats?period=24h" \
  -H "Authorization: Bearer YOUR_API_KEY"
```

```javascript JavaScript
const response = await fetch(
  'https://your-worker.workers.dev/api/v1/members/scrape/stats?period=7d',
  {
    headers: {
      'Authorization': 'Bearer YOUR_API_KEY'
    }
  }
);

const stats = await response.json();
console.log(`Success rate: ${(stats.successes / stats.executions * 100).toFixed(2)}%`);
```

</RequestExample>

<ResponseExample>

```json Success (200)
{
  "name": "scrape",
  "period": "24h",
  "executions": 1523,
  "successes": 1498,
  "failures": 25,
  "avgDuration": 847,
  "p95Duration": 2341,
  "cacheHitRate": 0.73
}
```

</ResponseExample>
