## What You'll Build

In this guide, you'll create a complete AI workflow that:
- Executes at the edge on Cloudflare Workers
- Orchestrates multiple members (LLMs, functions, APIs)
- Manages state across execution
- Deploys globally in seconds

**Time**: ~5 minutes

## Prerequisites

- Node.js 18+ installed
- Cloudflare account (free tier works)
- Wrangler CLI installed: `npm install -g wrangler`

## Step 1: Install Conductor

```bash
npm install @ensemble-edge/conductor
```

## Step 2: Create Your First Ensemble

Create a directory structure:

```bash
mkdir -p my-conductor-project/ensembles
cd my-conductor-project
```

Create your first ensemble:

```yaml
# ensembles/hello-world.yaml
name: hello-world
description: My first Conductor ensemble

flow:
  - member: greet
    type: Think
    config:
      provider: cloudflare
      model: '@cf/meta/llama-3.1-8b-instruct'
    input:
      prompt: Say hello to ${input.name} in a friendly way

output:
  greeting: ${greet.output.text}
```

## Step 3: Create Worker Entry Point

```typescript
// src/index.ts
import { ConductorRuntime } from '@ensemble-edge/conductor';

export default {
  async fetch(request: Request, env: Env, ctx: ExecutionContext) {
    const conductor = new ConductorRuntime({ env, ctx });

    const url = new URL(request.url);
    const name = url.searchParams.get('name') || 'World';

    const result = await conductor.executeEnsemble('hello-world', {
      input: { name }
    });

    return Response.json(result);
  }
};
```

## Step 4: Configure Wrangler

```toml
# wrangler.toml
name = "my-conductor-project"
main = "src/index.ts"
compatibility_date = "2024-01-01"

[ai]
binding = "AI"
```

## Step 5: Deploy to Cloudflare

```bash
# Login to Cloudflare
npx wrangler login

# Deploy
npx wrangler deploy
```

## Step 6: Test Your Ensemble

```bash
# Test with your deployed worker URL
curl "https://my-conductor-project.your-subdomain.workers.dev?name=World"
```

**Response:**
```json
{
  "success": true,
  "output": {
    "greeting": "Hello, World! Welcome to Conductor!"
  },
  "executionTime": 245,
  "memberCount": 1
}
```

## Next Steps

### Add More Members

```yaml
# ensembles/company-intel.yaml
name: company-intel
description: Analyze company data

state:
  schema:
    companyData: object
    analysis: object

flow:
  # Scrape company website
  - member: scrape
    type: API
    config:
      url: https://${input.domain}
    state:
      set: [companyData]

  # Analyze with AI
  - member: analyze
    type: Think
    config:
      provider: anthropic
      model: claude-3-5-sonnet-20241022
    input:
      prompt: Analyze this company data: ${state.companyData}
    state:
      set: [analysis]

  # Generate report
  - member: report
    type: Function
    code: |
      export default async function({ state }) {
        return {
          company: state.companyData.name,
          insights: state.analysis,
          generatedAt: new Date().toISOString()
        };
      }

output:
  report: ${report.output}
```

### Add State Management

State lets members share data without prop drilling:

```yaml
state:
  schema:
    userData: object
    processedData: object

flow:
  - member: fetch-user
    state:
      set: [userData]  # Write to state

  - member: process
    state:
      use: [userData]   # Read from state
      set: [processedData]  # Write result
```

### Add Parallel Execution

```yaml
flow:
  - parallel:
      - member: fetch-financials
      - member: fetch-news
      - member: fetch-competitors

  - member: combine-results
    input:
      financials: ${fetch-financials.output}
      news: ${fetch-news.output}
      competitors: ${fetch-competitors.output}
```

### Add Caching

```yaml
- member: expensive-api-call
  cache:
    ttl: 3600  # Cache for 1 hour
    key: ${input.id}
```

### Add Quality Scoring

```yaml
scoring:
  enabled: true
  defaultThresholds:
    minimum: 0.7
    target: 0.85

flow:
  - member: generate-content
    scoring:
      evaluator: validate
      thresholds:
        minimum: 0.8
      criteria:
        accuracy: "Content must be factually accurate"
      onFailure: retry
```

```bash
npm run dev

curl http://localhost:8787?name=World
```

## Testing Your Ensembles

```typescript
import { TestConductor } from '@ensemble-edge/conductor/testing';

describe('hello-world ensemble', () => {
  it('should greet user', async () => {
    const conductor = await TestConductor.create({
      projectPath: '.'
    });

    const result = await conductor.executeEnsemble('hello-world', {
      input: { name: 'Alice' }
    });

    expect(result).toBeSuccessful();
    expect(result.output.greeting).toContain('Alice');
  });
});
```
